# veloera
```shell
#!/bin/bash
@echo off

echo stopping veloera service
sudo systemctl stop veloera
echo stop succesfully

echo removing binary
rm -f veloera-linux-arm64
echo removing successfully

echo download latest binary
wget https://github.com/Veloera/Veloera/releases/latest/download/veloera-linux-arm64
echo download successfully

echo start veloera service
sudo systemctl start veloera
```
