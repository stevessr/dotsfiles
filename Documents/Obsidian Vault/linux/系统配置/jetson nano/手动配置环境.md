# 手动安装cuda相关
```shell
sudo apt install cuda-toolkit-10-2
sudo apt install nvidia-cudnn8
sudo apt install python3-pip
sudo pip3 install -U jetson-stats
sudo apt clean
sudo reboot
```
# Kubernetes
```shell
sudo apt update && sudo apt install -y apt-transport-https curl
curl -s https://packages.cloud.google.com/apt/doc/apt-key.gpg | sudo apt-key add -

#国内不要用这个
# Add the Kubernetes repo
cat <<EOF | sudo tee /etc/apt/sources.list.d/kubernetes.list
deb https://apt.kubernetes.io/ kubernetes-xenial main
EOF
```

```shell
#使用ustc的源
echo "deb [signed-by=/etc/apt/keyrings/kubernetes-apt-keyring.gpg] https://mirrors.ustc.edu.cn/kubernetes/core:/stable:/v1.28/deb/ /" | sudo tee /etc/apt/sources.list.d/kubernetes.list

curl -fsSL https://pkgs.k8s.io/core:/stable:/v1.28/deb/Release.key | sudo gpg --dearmor -o /etc/apt/keyrings/kubernetes-apt-keyring.gpg
```

# 关闭图形页面
```shell
sudo systemctl set-default multi-user.target

# ubuntu启用图形用户界面
sudo systemctl set-default graphical.target
sudo reboot

# 临时
sudo init 3 #关闭桌面
sudo init 5 #重启桌面
```

# u盘什么的要手动挂载
```shell
#先看u盘的名称
lsblk
sudo fdisk -l
sudo mkdir /media/usb
sudo mount /dev/sda1 /media/usb
```
